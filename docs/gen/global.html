applyWith(slotValuePairs)

update a group of slots by applying functions upon them, and starts a mutation proccess whose roots are these slots to be changed

Parameters:
Name Type Description
slotValuePairs array

each element is an array, whose first value is a Slot, and second is the function to be applied
Example
let $$p1 = $$(1).tag('p1');
let $$p2 = $$(2).tag('p2');
let $$p3 = $$p2.fork(it => it + 1).tag('p3');
let $$p4 = $$(function ([p1, p2, p3], { roots, involved }) {
  console.log(roots.map(it => it.tag())); // p1, p2
  console.log(involved.map(it => it.tag())); // p1, p2, p3
  return p1 + p2 + p3;
}, [$$p1, $$p2, $$p3]);
$$.applyWith([
  [$$p1, n => n + 1],
  [$$p2, n => n + 2],
]);
console.log($$p1.val(), $$p2.val(), $$p3.val(), $$p4.val()); // 2, 4, 5, 11
